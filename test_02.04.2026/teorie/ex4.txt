## Ce valori for fi afisate?

#include <iostream
using namespace std;

int main() {
  int a[100] = {0, 1, 2, 3, 4, 5, 6, 7, 8}; // &a[0] = 0x00
  int* ptr;

  ptr = a;

  cout << ptr << '\n\;

  cout << ptr + 1 << '\n';
  cout << *(ptr + 1) << '\n';
  cout << *(ptr) + 1 << '\n\;
  
  *(ptr + 3) = 4;
  cout << a[3] << '\n';
  return 0;
}
